# IO

IO是指Input/Output，即输入和输出。以内存数据为中心：

  • Input指从外部读入数据到内存，例如，把文件从磁盘读取到内存，从网络读取数据到内存等等。

  • Output指把数据从内存输出到外部，例如，把数据从内存写入到文件，把数据从内存输出到网络等等

IO流是一种流式的数据输入/输出模型：

  • 字节流 : 二进制数据以byte为最小单位在InputStream/OutputStream中单向流动；

  • 字符流 : 字符数据以char为最小单位在Reader/Writer中单向流动

  • 缓冲流: 就是缓冲字节或者字符的流,效率高

上面是一些常识,其实我对IO 的理解就是找对 数据源 和 数据目的地,剩下的就是读写取数据了 一般来说,input 是读,output 是写.那读写的内容就是数据流,根据数据流不同分为字节流(bytes),字符流(chars),然后根据效率有了缓冲流

下面具体来说明

public class FileIoDemo {
    
    File file;
    
    //字节输入流
    FileInputStream in;
    
    //字节输出流
    FileOutputStream out;
    
    
    //字符输入流
    FileReader reader;
    
    //字符输出流
    FileWriter writer;
    
    //字节缓冲流
    BufferedInputStream bufferedInputStream;
    
    BufferedOutputStream bufferedOutputStream;
    
    //字符缓冲流
    BufferedReader bufferedReader;
    
    BufferedWriter bufferedWriter;
    
    public static void main(String[] args) throws IOException {
        String lineSeparator = System.getProperty("line.separator");
        System.out.println("换行符." + lineSeparator);
        
        FileIoDemo f = new FileIoDemo();
        f.initFile();
        //创建
        f.createFile();
        f.exists();
        //删除
        f.deleteFile();
        f.exists();
        //创建
        f.createFile();
        //获取绝对路径
        f.getAbsolutePath();
        // 字节流写
        f.writeFileByByte("字节流写" + lineSeparator);
        //字节流读
        f.readFileByByte();
        //字符流写
        f.writeFileByChar("字符流写" + lineSeparator);
        //字符流读
        f.readFileByChar();
        //字节缓冲流写
        f.writeFileByBuffByte("字节缓冲流写" + lineSeparator);
        //字节缓冲流读
        f.readFileByBuffByte();
        //字符缓冲流写
        f.writeFileByBuffChar("字符缓冲流写" + lineSeparator);
        //字符缓冲流读
        f.readFileByBuffChar();
        System.out.println();
        System.out.println("************************一次性读取整个文件内容");
        f.readFileByBuffChar();
    }
    
    public void initFile() {
        //直接传入文件地址，来创建File对象,这里可以是目录或者文件
        String fileParentPath = "E:/test";
        String fileChildPath = "1.txt";
        file = new File(fileParentPath, fileChildPath);
        String filePath = "E:/test/1.txt";
        file = new File(filePath);
    }
    
    public boolean exists() {
        System.out.println("文件是否存在: " + file.exists());
        System.out.println("文件是目录? : " + file.isDirectory());
        return file == null ? false : file.exists();
    }
    
    public void createFile() throws IOException {
        if (!file.exists()) {
            file.getParentFile().mkdirs();
            file.createNewFile();
        }
    }
    
    public String getAbsolutePath() {
        String absolutePath = file.getAbsolutePath();
        System.out.println("文件绝对路径是: " + absolutePath);
        return absolutePath;
    }
    
    public void deleteFile() throws IOException {
        file.delete();
    }
    
    /**
     * 字节流写文件
     */
    public void writeFileByByte(String text) throws IOException {
        out = new FileOutputStream(file, true);
        out.write(text.getBytes(StandardCharsets.UTF_8));
        out.flush();
        out.close();
    }
    
    /**
     * 字节流读文件
     */
    public void readFileByByte() throws IOException {
        in = new FileInputStream(file);
        int len = 0;
        byte[] buffer = new byte[1024];
        while ((len = in.read(buffer)) != -1) {
            System.out.print(new String(buffer, 0, len));
        }
        in.close();
    }
    
    /**
     * 字符流读文件
     */
    public void readFileByChar() throws IOException {
        reader = new FileReader(file);
        int len = 0;
        char[] buffer = new char[1024];
        while ((len = reader.read(buffer)) != -1) {
            System.out.print(new String(buffer, 0, len));
        }
        in.close();
    }
    
    /**
     * 字符流写文件
     */
    public void writeFileByChar(String text) throws IOException {
        writer = new FileWriter(file, true);
        char[] charBuff = text.toCharArray();
        writer.write(charBuff);
        writer.flush();
        writer.close();
    }
    
    /**
     * 字节缓冲流读文件
     */
    public void readFileByBuffByte() throws IOException {
        bufferedInputStream = new BufferedInputStream(new FileInputStream(file));
        int len = 0;
        byte[] buffer = new byte[1024];
        while ((len = bufferedInputStream.read(buffer)) != -1) {
            System.out.print(new String(buffer, 0, len));
        }
        bufferedInputStream.close();
    }
    
    /**
     * 字节缓冲流写文件
     */
    public void writeFileByBuffByte(String text) throws IOException {
        bufferedOutputStream = new BufferedOutputStream(new FileOutputStream(file, true));
        byte[] bytes = text.getBytes(StandardCharsets.UTF_8);
        bufferedOutputStream.write(bytes);
        bufferedOutputStream.flush();
        bufferedOutputStream.close();
    }
    
    /**
     * 字符缓冲流读文件
     */
    public void readFileByBuffChar() throws IOException {
        bufferedReader = new BufferedReader(new FileReader(file));
        char[] chars = new char[1024];
        int len = 0;
        while ((len = bufferedReader.read(chars)) != -1) {
            System.out.print(new String(chars, 0, len));
        }
        bufferedReader.close();
    }
    
    /**
     * 字符缓冲流写文件
     */
    public void writeFileByBuffChar(String text) throws IOException {
        bufferedWriter = new BufferedWriter(new FileWriter(file, true));
        bufferedWriter.write(text);
        bufferedWriter.flush();
        bufferedWriter.close();
    }
}
