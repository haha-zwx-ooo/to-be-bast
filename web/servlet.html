# servlet

Servlet 是一个在 Web 服务器上的 JVM 内运行的 java 程序。它用于开发动态 Web 应用。

# 案例

本文以tomcat为容器举例,来说明servlet使用和原理.为了便捷,本文使用嵌入式tomcat容器

  1. 首先声明tomcat容器

@Data
public class TomcatServer {
    
    protected static volatile Tomcat tomcat;
    
    protected Context context;
    
    private int port = 8080;
    
    private String contextPath = "/";
    
    public void addServlet(String servletContextPath, String servletName, String servletClass) throws ServletException {
        init();
        tomcat.addServlet(contextPath, servletName, servletClass);
        context.addServletMapping(servletContextPath, servletName);
    }
    
    private void init() throws ServletException {
        if (tomcat == null) {
            synchronized (TomcatServer.class) {
                if (tomcat == null) {
                    tomcat = new Tomcat();
                    Connector connector = new Connector();
                    connector.setPort(port);
                    connector.setURIEncoding("UTF-8");
                    connector.setAsyncTimeout(2000);
                    tomcat.getService().addConnector(connector);
                    tomcat.setConnector(connector);
                    context = tomcat.addContext(contextPath, new File(".").getAbsolutePath());
                }
            }
        }
    }
    
    public void start() throws ServletException, LifecycleException {
        init();
        tomcat.enableNaming();
        tomcat.start();
        tomcat.getServer().await();
    }
}
  1. 编写servlet 并执行
public class HelloServlet extends HttpServlet {
    
    @Override
    protected void service(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        String a = req.getParameter("a");
        System.out.println("service方法调用");
        resp.setContentType("text/html;charset=UTF-8");
        resp.getWriter().println("service方法调用 调用参数 a : " + a);
        super.service(req, resp);
    }
    
    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        System.out.println("doGet方法调用");
        resp.setContentType("text/html;charset=UTF-8");
        resp.getWriter().println("doGet方法调用");
    }
    
    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        System.out.println("doPost方法调用");
        resp.setContentType("text/html;charset=UTF-8");
        resp.getWriter().println("doPost方法调用");
    }
}
  1. 执行
public class Main {
    
    public static void main(String[] args) throws ServletException, LifecycleException {
        TomcatServer server = new TomcatServer();
        server.setPort(8081);
        server.setContextPath("/");
        server.addServlet("/hello", "HelloServlet", "com.haha.servlet.api.HelloServlet");
        server.start();
    }
}

# 核心原理

整个servlet 我们重点关心的是 Servlet 接口 ,其中 service方法是重点

public interface Servlet {
    
    void init(ServletConfig var1) throws ServletException;
    
    ServletConfig getServletConfig();
    
    void service(ServletRequest var1, ServletResponse var2) throws ServletException, IOException;
    
    String getServletInfo();
    
    void destroy();
}

HttpServlet 的 service方法,这个方法相当于 处理所有HttpServletRequest 的前置入口

public abstract class HttpServlet extends GenericServlet {
    
    protected void service(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        String method = req.getMethod();
        long lastModified;
        if (method.equals("GET")) {
            lastModified = this.getLastModified(req);
            if (lastModified == -1L) {
                this.doGet(req, resp);
            } else {
                long ifModifiedSince = req.getDateHeader("If-Modified-Since");
                if (ifModifiedSince < lastModified) {
                    this.maybeSetLastModified(resp, lastModified);
                    this.doGet(req, resp);
                } else {
                    resp.setStatus(304);
                }
            }
        } else if (method.equals("HEAD")) {
            lastModified = this.getLastModified(req);
            this.maybeSetLastModified(resp, lastModified);
            this.doHead(req, resp);
        } else if (method.equals("POST")) {
            this.doPost(req, resp);
        } else if (method.equals("PUT")) {
            this.doPut(req, resp);
        } else if (method.equals("DELETE")) {
            this.doDelete(req, resp);
        } else if (method.equals("OPTIONS")) {
            this.doOptions(req, resp);
        } else if (method.equals("TRACE")) {
            this.doTrace(req, resp);
        } else {
            String errMsg = lStrings.getString("http.method_not_implemented");
            Object[] errArgs = new Object[] {method};
            errMsg = MessageFormat.format(errMsg, errArgs);
            resp.sendError(501, errMsg);
        }
    
    }
}
